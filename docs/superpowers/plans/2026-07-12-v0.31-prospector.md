# v0.31 "Prospector" Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship the /map reveal-radius fix + /map zoom + the Item Filter engine (storage, editor, ground-item + character-equipment highlight surfaces) so users can flag desired affix combinations and see matching items lit up in-game.

**Architecture:** New `ItemFilterEngine` in `POE2Radar.Core.Game` holds a list of filter rules loaded from `config/item_filters.json`. `RadarApp` calls the engine per world tick against ground items (extend `BuildItemLabels`) and equipped items (new `Poe2Character.ReadEquipmentSlotRects` + new draw method). Dashboard gains an "Item Filters" tab (card grid). /map JS gets a bigger reveal disc + mousewheel/keyboard zoom. All mutations are cosmetic — the tool's read-only compliance envelope is unchanged.

**Tech Stack:** C# 12 / .NET 10 (net10.0-windows) · xUnit 2 · Direct2D via Vortice · vanilla JS (map.js, DashboardHtml.cs inline scripts) · bd 1.1.0 + opencode + OpenRouter/qwen3-coder for mechanical beads.

## Global Constraints

Every task inherits these — copied verbatim from spec §3:

- **Read-only compliance envelope**: no memory writes, no input injection, no `Marshal.Write*` / `WriteProcessMemory` / `SendInput` / `keybd_event`. Enforced by `scripts/compliance-gate.ps1` which MUST PASS.
- **No feature gates on functional behavior**: filter engine is cosmetic-highlight only; NEVER hides / blocks / replaces game UI.
- **Zero perf impact when idle**: filter matching runs only when affix reads already happen. Dashboard-side match counter runs on SSE snapshots.
- **String scrubber**: `scripts/scrub-strings.ps1 -SelfTest` MUST PASS on the release build — no internal-tooling leaks in new strings.
- **Attribution sentinel**: `scripts/attribution-sentinel-gate.ps1` (Draft) MUST PASS — no reference to `superpowers/`, `SDD`, `Sikaka`, `GameHelper` in user-visible strings, CHANGELOG, in-app text, or release notes.
- **Backwards-compat**: an existing user with no `item_filters.json` sees zero behavior change until they enable a filter. Presets ship `enabled: false` on first-run copy.
- **PRESERVE ALL OTHER BEHAVIOR**: every task's scope is exactly what its Files block lists. No unrelated cleanup, no drive-by refactors.
- **Build + test on every task**: `dotnet build POE2Radar.slnx -c Release --nologo` (0 errors) + `dotnet test POE2Radar.slnx -c Release --nologo --no-build` (all pre-existing tests still pass + new tests introduced by the task pass).

/// <summary>
/// v0.31 Prospector: user's item filter ruleset — persistent JSON-backed list of highlight
/// rules that match affix combinations on items. Load-on-construct + save-on-mutate, mirroring
/// the DisplayRules pattern (lock + immutable snapshot + generation counter). Empty file /
/// malformed file → empty list; first run copies the shipped starter presets (all disabled).
/// The Match method (T6) is the render-hot path; keep it allocation-free per call.
/// </summary>
public sealed class ItemFilterEngine
{
    private readonly string _filePath;
    private readonly object _gate = new();
    private List<FilterRule> _rules = new();
    private int _generation;

    private static readonly JsonSerializerOptions Json = new()
    {
        WriteIndented = true,
        PropertyNamingPolicy = JsonNamingPolicy.CamelCase,
        DefaultIgnoreCondition = JsonIgnoreCondition.WhenWritingNull,
    };

    public ItemFilterEngine(string filePath)
    {
        _filePath = filePath;
        Load();
    }

    public int Generation { get { lock (_gate) return _generation; } }
    public IReadOnlyList<FilterRule> All { get { lock (_gate) return _rules.ToList(); } }

    public void Replace(IEnumerable<FilterRule> rules)
    {
        lock (_gate) { _rules = rules.ToList(); _generation++; Save(); }
    }
    public void Add(FilterRule rule)
    {
        lock (_gate) { _rules.Add(rule); _generation++; Save(); }
    }
    public void RemoveAt(int index)
    {
        lock (_gate)
        {
            if (index < 0 || index >= _rules.Count) return;
            _rules.RemoveAt(index); _generation++; Save();
        }
    }
    public void Update(int index, FilterRule rule)
    {
        lock (_gate)
        {
            if (index < 0 || index >= _rules.Count) return;
            _rules[index] = rule; _generation++; Save();
        }
    }
    public void Move(int fromIndex, int toIndex)
    {
        lock (_gate)
        {
            if (fromIndex < 0 || fromIndex >= _rules.Count) return;
            toIndex = Math.Clamp(toIndex, 0, _rules.Count - 1);
            if (fromIndex == toIndex) return;
            var r = _rules[fromIndex];
            _rules.RemoveAt(fromIndex);
            _rules.Insert(toIndex, r);
            _generation++; Save();
        }
    }
    public void RestoreStarterPresets()
    {
        var seed = LoadEmbeddedPresets();
        lock (_gate)
        {
            var existingIds = new HashSet<string>(_rules.Select(r => r.Id), StringComparer.Ordinal);
            foreach (var p in seed)
                if (!existingIds.Contains(p.Id)) _rules.Add(p);
            _generation++; Save();
        }
    }

    private void Load()
    {
        try
        {
            if (!File.Exists(_filePath))
            {
                var seed = LoadEmbeddedPresets();
                _rules = seed.ToList();
                Save();   // materialize seeds on disk
                return;
            }
            var doc = JsonSerializer.Deserialize<FilterFile>(File.ReadAllText(_filePath), Json);
            if (doc?.Filters is not null) _rules = doc.Filters.ToList();
        }
        catch (Exception ex) { Console.Error.WriteLine($"Item filters load failed: {ex.Message}"); }
    }

    private void Save()
    {
        try
        {
            var dir = Path.GetDirectoryName(_filePath);
            if (!string.IsNullOrEmpty(dir)) Directory.CreateDirectory(dir);
            File.WriteAllText(_filePath, JsonSerializer.Serialize(new FilterFile(_rules.ToList()), Json));
        }
        catch (Exception ex) { Console.Error.WriteLine($"Item filters save failed: {ex.Message}"); }
    }

    /// <summary>Read the shipped default_item_filters.json embedded resource; empty list on
    /// missing / malformed resource (fail-safe).</summary>
    public static IReadOnlyList<FilterRule> LoadEmbeddedPresets()
    {
        try
        {
            var asm = Assembly.GetExecutingAssembly();
            var name = asm.GetManifestResourceNames().FirstOrDefault(n => n.Contains("default_item_filters", StringComparison.Ordinal));
            if (name is null) return Array.Empty<FilterRule>();
            using var s = asm.GetManifestResourceStream(name);
            if (s is null) return Array.Empty<FilterRule>();
            var doc = JsonSerializer.Deserialize<FilterFile>(s, Json);
            return doc?.Filters ?? Array.Empty<FilterRule>();
        }
        catch (Exception ex) { Console.Error.WriteLine($"Item filter presets load failed: {ex.Message}"); return Array.Empty<FilterRule>(); }
    }

    private sealed record FilterFile([property: JsonPropertyName("filters")] IReadOnlyList<FilterRule> Filters);
}

```csharp
public sealed record MatchedFilter(FilterRule Rule, string Reason);

// Add these using directives at the top of the file if not already present:
// using static POE2Radar.Core.Game.Poe2Live;

public IReadOnlyList<MatchedFilter> Match(
    IReadOnlyList<Poe2Live.RawAffix> implicit_,
    IReadOnlyList<Poe2Live.RawAffix> explicit_,
    ModRanges? modRanges = null,
    ItemModTranslator? translator = null)
{
    FilterRule[] snapshot;
    lock (_gate) snapshot = _rules.ToArray();

    var matched = new List<MatchedFilter>();
    for (var i = 0; i < snapshot.Length; i++)
    {
        var rule = snapshot[i];
        if (!rule.Enabled) continue;
        if (rule.Requirements is not { Count: > 0 }) continue;
        if (RuleMatches(rule, implicit_, explicit_, modRanges, translator, out var reason))
            matched.Add(new MatchedFilter(rule, reason));
    }
    matched.Sort((a, b) => b.Rule.Priority.CompareTo(a.Rule.Priority));
    return matched;
}

private static bool RuleMatches(
    FilterRule rule,
    IReadOnlyList<Poe2Live.RawAffix> implicit_,
    IReadOnlyList<Poe2Live.RawAffix> explicit_,
    ModRanges? modRanges,
    ItemModTranslator? translator,
    out string reason)
{
    var passed = 0;
    foreach (var req in rule.Requirements)
    {
        if (!RequirementSatisfied(req, implicit_, explicit_, modRanges, translator))
        {
            reason = $"failed req {passed + 1}/{rule.Requirements.Count} ({req.StatId} {req.Op} {req.Value})";
            return false;
        }
        passed++;
    }
    reason = $"passed {passed}/{rule.Requirements.Count} reqs";
    return true;
}

private static bool RequirementSatisfied(
    FilterRequirement req,
    IReadOnlyList<Poe2Live.RawAffix> implicit_,
    IReadOnlyList<Poe2Live.RawAffix> explicit_,
    ModRanges? modRanges,
    ItemModTranslator? translator)
{
    // Test-friendly fallback: if translator is null, treat req.StatId as a substring probe on
    // the mod id. Real callers pass the translator so we route via stable stat ids.
    foreach (var aff in EnumerateCandidates(req, implicit_, explicit_, modRanges, translator))
    {
        foreach (var v in aff.Values ?? Array.Empty<int>())
            if (OpMatches(req, v)) return true;
    }
    return false;
}

private static IEnumerable<Poe2Live.RawAffix> EnumerateCandidates(
    FilterRequirement req,
    IReadOnlyList<Poe2Live.RawAffix> implicit_,
    IReadOnlyList<Poe2Live.RawAffix> explicit_,
    ModRanges? modRanges,
    ItemModTranslator? translator)
{
    var scope = req.Scope;
    bool wantImplicit = scope is null || scope.Any(s => string.Equals(s, "implicit", StringComparison.OrdinalIgnoreCase));
    bool wantPrefix   = scope is null || scope.Any(s => string.Equals(s, "prefix",   StringComparison.OrdinalIgnoreCase));
    bool wantSuffix   = scope is null || scope.Any(s => string.Equals(s, "suffix",   StringComparison.OrdinalIgnoreCase));
    if (wantImplicit) foreach (var a in implicit_) if (MatchesStat(a, req, translator) && WithinTier(a, req, modRanges)) yield return a;
    if (wantPrefix || wantSuffix)
        foreach (var a in explicit_)
            if (MatchesStat(a, req, translator) && WithinTier(a, req, modRanges) && MatchesSlot(a, req, translator))
                yield return a;
}

private static bool MatchesStat(Poe2Live.RawAffix aff, FilterRequirement req, ItemModTranslator? translator)
{
    if (translator is null)
    {
        // Fallback: modId substring match on StatId. Test-friendly + tolerates translator not
        // being wired at engine call sites that don't need scope/tier semantics.
        return aff.ModId?.Contains(req.StatId, StringComparison.OrdinalIgnoreCase) == true;
    }
    var statIds = translator.StatIdsFor(aff.ModId);
    if (statIds is null || statIds.Count == 0) return false;
    foreach (var sid in statIds)
        if (string.Equals(sid, req.StatId, StringComparison.Ordinal)) return true;
    return false;
}

private static bool MatchesSlot(Poe2Live.RawAffix aff, FilterRequirement req, ItemModTranslator? translator)
{
    if (req.Scope is null || translator is null) return true;
    var slot = translator.SlotOf(aff.ModId) ?? "";
    return req.Scope.Any(s => string.Equals(s, slot, StringComparison.OrdinalIgnoreCase));
}

private static bool WithinTier(Poe2Live.RawAffix aff, FilterRequirement req, ModRanges? modRanges)
{
    if (req.MaxTier is not int maxT || modRanges is null) return true;
    var tier = modRanges.TierFor(aff.ModId);
    return tier is null || tier.Value <= maxT;
}

private static bool OpMatches(FilterRequirement req, double v)
{
    return req.Op switch
    {
        ">=" => v >= req.Value,
        "<=" => v <= req.Value,
        "==" => Math.Abs(v - req.Value) < 1e-9,
        "between" => v >= req.Value && v <= (req.ValueMax ?? req.Value),
        _ => false,
    };
}
```

**Note on `ItemModTranslator.SlotOf` and `ModRanges.TierFor`**: these signatures MUST exist. If they don't, the implementer adds thin wrappers in the respective classes returning `null` on unknown mod ids (fail-safe — the requirement then behaves as any-slot / any-tier).

```csharp
namespace POE2Radar.Core.Game;

using System.Numerics;
using Vortice;

public enum EquipmentSlot { Weapon1, Offhand1, Weapon2, Offhand2, Chest, Helmet, Gloves, Boots, Belt, Amulet, Ring1, Ring2, Flask1, Flask2, Flask3, Flask4, Flask5 }

public readonly record struct EquipmentSlotFingerprint(
    EquipmentSlot Slot,
    int ExpectedChildIndex,
    float ExpectedRelX,
    float ExpectedRelY,
    float ExpectedRelW,
    float ExpectedRelH);

/// <summary>
/// v0.31 Prospector: character equipment panel walker. Locates the panel via
/// <see cref="Poe2Offsets.CharacterPanel.UiRootChildIndex"/>, then finds each equipment slot's
/// visible screen rect using fingerprint backtracking (tolerant to child-index drift within
/// the same panel structure). Returns empty when the panel is closed / not visible.
/// Read-only.
/// </summary>
public sealed class Poe2Character
{
    private readonly MemoryReader _reader;

    public Poe2Character(MemoryReader reader) { _reader = reader; }

    public IReadOnlyDictionary<EquipmentSlot, RawRectF> ReadEquipmentSlotRects(nint uiRoot, int winW, int winH)
    {
        var results = new Dictionary<EquipmentSlot, RawRectF>();

        var panel = ResolveCharacterPanel(uiRoot);
        if (panel == 0) return results;
        if (!IsVisible(panel)) return results;

        // Read the panel rect for relative-position math.
        if (!TryUiElementRect(panel, winW, winH, out var panelRect)) return results;
        var panelW = panelRect.Right - panelRect.Left;
        var panelH = panelRect.Bottom - panelRect.Top;
        if (panelW < 1f || panelH < 1f) return results;

        // Enumerate panel children (BFS one level deep).
        var childCount = _reader.Read<int>(panel + Poe2Offsets.UiElement.Children /* +0xC8-ish; see existing usage */);
        var childrenPtr = _reader.Read<nint>(panel + Poe2Offsets.UiElement.Children);
        // (Actual field layout: consult Poe2Live.WalkUiTree for the canonical children walk. Reuse
        // its helper rather than re-implementing the vector walk.)

        var children = Poe2Live.EnumerateChildren(_reader, panel);   // helper — grep Poe2Live for the pattern

        foreach (var fp in Poe2Offsets.CharacterPanel.SlotFingerprints)
        {
            // Try the expected index first.
            nint slotElement = 0;
            if (fp.ExpectedChildIndex >= 0 && fp.ExpectedChildIndex < children.Count)
            {
                var candidate = children[fp.ExpectedChildIndex];
                if (Fingerprints(candidate, fp, winW, winH, panelRect)) slotElement = candidate;
            }
            // Fallback: linear backtracking scan.
            if (slotElement == 0)
                foreach (var c in children)
                    if (Fingerprints(c, fp, winW, winH, panelRect)) { slotElement = c; break; }

            if (slotElement == 0) continue;
            if (!TryUiElementRect(slotElement, winW, winH, out var slotRect)) continue;
            results[fp.Slot] = slotRect;
        }
        return results;
    }

    private nint ResolveCharacterPanel(nint uiRoot)
    {
        var childIdx = Poe2Offsets.CharacterPanel.UiRootChildIndex;
        var children = Poe2Live.EnumerateChildren(_reader, uiRoot);
        return childIdx >= 0 && childIdx < children.Count ? children[childIdx] : 0;
    }

    private bool IsVisible(nint element)
    {
        var flags = _reader.Read<uint>(element + Poe2Offsets.UiElement.Flags);
        return (flags & (1u << 0x0B)) != 0;   // spec §3 visible-bit
    }

    private bool TryUiElementRect(nint element, int winW, int winH, out RawRectF rect)
    {
        // Reuse the existing helper — grep Poe2Live.cs:1251 for TryUiElementRect signature.
        return Poe2Live.TryUiElementRect(_reader, element, winW, winH, out rect);
    }

    private bool Fingerprints(nint element, EquipmentSlotFingerprint fp, int winW, int winH, RawRectF panelRect)
    {
        if (!TryUiElementRect(element, winW, winH, out var rect)) return false;
        var w = rect.Right - rect.Left;
        var h = rect.Bottom - rect.Top;
        var panelW = panelRect.Right - panelRect.Left;
        var panelH = panelRect.Bottom - panelRect.Top;
        if (w < 1f || h < 1f || panelW < 1f || panelH < 1f) return false;
        var relX = (rect.Left - panelRect.Left) / panelW;
        var relY = (rect.Top - panelRect.Top) / panelH;
        var relW = w / panelW;
        var relH = h / panelH;
        const float tol = 0.10f;   // 10% tolerance for backtracking
        return Math.Abs(relX - fp.ExpectedRelX) < tol
            && Math.Abs(relY - fp.ExpectedRelY) < tol
            && Math.Abs(relW - fp.ExpectedRelW) < tol
            && Math.Abs(relH - fp.ExpectedRelH) < tol;
    }
}
```

Adjust helper names to match the actual `Poe2Live` API — if `EnumerateChildren` doesn't exist there, use whatever the codebase's canonical BFS walker exposes.
