# POE2GPS v0.21 "Guided Campaign" Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship a two-track release: (Track A) integrate syrairc/ExileCampaigns2 route data + advance logic as a new step-by-step Guided Campaign layer alongside existing Campaign GPS; (Track B) harden the community pipeline with buff + preload Contribute rails, Worker anti-abuse, issue templates, credit loop, and merge-script consolidation.

**Architecture:** Parallel Rail — EC2 AdvanceEngine + WorldStateAdapter + RouteCursor run alongside (not replacing) ZoneOrderProgress + CampaignGps + ObjectiveDirector, hooked at `RadarApp.CampaignReconcile:2493` and publishing to a new additive `CampaignGuide` SSE key. Six inlined stub methods (`QuestFlagSatisfied`, `WaypointPulsed`, `SatisfiedFlagCount`, `TalkProgress`, `InteractProgress`, quest-inventory `LootSatisfied`) return `false/0`; graceful area-boundary forward-snap advances past flag-only stalled steps. Cloudflare Worker splits into three sibling routes with shared NFKD leet-fold profanity + KV-counter rate-limit middleware, deploys to prod BEFORE the app build ships with new Contribute buttons.

**Tech Stack:** C# / .NET 8 / net8.0-windows / Direct2D + browser Canvas 2D + SSE; Cloudflare Workers + KV; Python 3 (merge scripts); PowerShell 5.1 (compliance-gate + release scripts); bash (smoke script + relabel script).

## Global Constraints

- **Zero memory writes.** Every new memory access is read-only over `IngameState.Data.ServerData` or the shipped `PlayerInventories` chain. No input injection, no telemetry, no new egress except existing update check.
- **Zero-cost-when-off.** `WorldStateAdapter` and `RouteCursor` respect `_settings.EnableCampaignGps`; adapter constructed unconditionally at startup so runtime toggle cannot race lazy init; `Tick` gated on the setting.
- **v0.20 wire-format compat.** New SSE data lands as additive keys only. No renames. No removals. Verify gate: v0.20.1 client sniff transcript + golden DTO snapshot test.
- **No internal-tooling paths in public surfaces.** CI grep gate rejects superpowers-related paths in README, CHANGELOG, `.github/ISSUE_TEMPLATE/*.yml`, root `CONTRIBUTING.md`, `docs/community-pipeline.md`, Worker HTML, any in-app string.
- **Attribution sentinel gate.** During DRAFT phase (tasks 1-8), `TODO(syrairc-license)` + `TODO(syrairc-hash)` sentinels are the only permitted placeholder tokens in ported files. At FORMALIZE (task 9) the gate flips: reject any surviving `TODO(syrairc-*)` sentinel.
- **syrairc DM hard-gates the tag (PMS-12).** No attribution-lite fallback — HOLD tag if DM stalls.
- **Worker deploy ordering (PMS-13).** CF-WORKER must deploy to Cloudflare prod (smoke-verified) BEFORE CF-DASH-BUTTONS PR opens.
- **SL #9 TieredCatalog shipping assumption:** grep-check before starting CF-DASH-BUTTONS. If not shipped, defer SL #10 buff-fold to v0.22.
- All tests via `dotnet test` from repo root; new tests live under `tests/POE2Radar.Tests/`.
- Frequent commits. One commit per task (see Step N of each task); no cross-task WIP commits.

      string? ImportFp);

  {

      public static readonly RouteDocument Empty = new(CurrentVersion, new List<RouteStep>());

      public static RouteDocument Load(Stream stream)
      {
          if (stream is null) throw new ArgumentNullException(nameof(stream));
          var opts = new JsonSerializerOptions
          {
              PropertyNameCaseInsensitive = true,
              ReadCommentHandling = JsonCommentHandling.Skip,
              AllowTrailingCommas = true,
              Converters = { new JsonStringEnumConverter(JsonNamingPolicy.CamelCase, allowIntegerValues: true) }
          };
          RouteDocument? doc;
          try { doc = JsonSerializer.Deserialize<RouteDocument>(stream, opts); }
          catch (JsonException ex)
          {
              throw new RouteSchemaMismatchException("route.json failed to deserialize", ex);
          }
          if (doc is null) throw new RouteSchemaMismatchException("route.json deserialized to null");
          if (doc.Version != CurrentVersion)
              throw new RouteSchemaMismatchException($"route.json schemaVersion={doc.Version}, expected {CurrentVersion}");
          return doc;
      }

      public RouteSchemaMismatchException(string message, Exception inner) : base(message, inner) { }

  // Ported from ExileCampaigns2 by syrairc under TODO(syrairc-license) — upstream commit TODO(syrairc-hash)
  using System;
  using System.Text.RegularExpressions;

  namespace POE2Radar.Core.Campaign.Guide;

  public sealed class PatternMatcher
  {
      private readonly Pattern _pattern;
      private readonly Regex? _regex;
      public string? Error { get; }

      public PatternMatcher(Pattern pattern)
      {
          _pattern = pattern ?? new Pattern("");
          if (_pattern.Regex && !string.IsNullOrEmpty(_pattern.Value))
          {
              try { _regex = new Regex(_pattern.Value, RegexOptions.IgnoreCase | RegexOptions.Compiled); }
              catch (ArgumentException ex) { _regex = null; Error = ex.Message; }
          }
      }

      public bool IsMatch(string? candidate)
      {
          if (string.IsNullOrEmpty(candidate)) return false;
          if (_regex != null) return _regex.IsMatch(candidate);
          if (string.IsNullOrEmpty(_pattern.Value)) return false;
          return candidate.IndexOf(_pattern.Value, StringComparison.OrdinalIgnoreCase) >= 0;
      }
  }

  using System;
  using System.Collections.Generic;

  namespace POE2Radar.Core.Campaign.Guide;

  // the only seam between the pure advance logic and live game memory. WorldStateAdapter (EC2-STATE)
  // implements this against Poe2Live snapshots; tests fake it.
  public interface IWorldState
  {
      bool QuestFlagSatisfied(Pattern flag);
      bool InAreaSatisfied(Pattern area);
      bool WaypointPulsed();
      bool ProximitySatisfied(IReadOnlyList<EntityMatcher>? entities, IReadOnlyList<Pattern>? tiles, float distance);
      bool LootSatisfied(IReadOnlyList<ItemMatcher>? items);
      int KillProgress(IReadOnlyList<EntityMatcher>? entities);
      int InteractProgress(IReadOnlyList<EntityMatcher>? entities);
      int TalkProgress(IReadOnlyList<EntityMatcher>? entities);
      int SatisfiedFlagCount(IReadOnlyList<Pattern> flags);
  }

  // evaluates whether the current step is done. the single advance gate.
  public static class AdvanceEngine
  {
      public const float DefaultProximity = 40f;

      public static bool IsStepComplete(RouteStep step, IWorldState world)
      {
          if (step?.Objectives is null || step.Objectives.Count == 0) return false;
          if (step.CompleteWhen == CompleteWhen.Any)
          {
              foreach (var o in step.Objectives)
                  if (ObjectiveComplete(o, world)) return true;
              return false;
          }
          foreach (var o in step.Objectives)
              if (!ObjectiveComplete(o, world)) return false;
          return true;
      }

      public static bool ObjectiveComplete(Objective o, IWorldState world) => o.Type switch
      {
          ObjectiveType.Kill             => CountDone(o, world, world.KillProgress),
          ObjectiveType.Interact         => CountDone(o, world, world.InteractProgress),
          ObjectiveType.Talk             => CountDone(o, world, world.TalkProgress),
          ObjectiveType.Loot             => world.LootSatisfied(o.Items),
          ObjectiveType.Proximity        => world.ProximitySatisfied(o.Entities, null, o.Distance > 0f ? o.Distance : DefaultProximity),
          ObjectiveType.QuestFlag        => o.Flag != null && world.QuestFlagSatisfied(o.Flag),
          ObjectiveType.EnterArea        => o.AreaTarget != null && world.InAreaSatisfied(o.AreaTarget),
          ObjectiveType.ActivateWaypoint => world.WaypointPulsed(),
          ObjectiveType.Manual           => false,
          _                              => false,
      };

      // Kill/Interact/Talk: progress flags win when present (e.g. the 3 obelisks), else the live counter.
      private static bool CountDone(Objective o, IWorldState world, Func<IReadOnlyList<EntityMatcher>?, int> liveCount)
      {
          int needed = o.Count > 0 ? o.Count : 1;
          int done = o.ProgressFlags is { Count: > 0 } ? world.SatisfiedFlagCount(o.ProgressFlags) : liveCount(o.Entities);
          return done >= needed;
      }
  }

  using System;

  namespace POE2Radar.Core.Campaign.Guide;

  /// <summary>Immutable per-tick snapshot of where the guide cursor is right now.
  /// Serialised into the additive <c>CampaignGuide</c> SSE key. Never mutated after publish.</summary>
  public readonly record struct CampaignStepInstruction(
      string StepId,
      string AreaId,
      int Act,
      int Ordinal,
      string Text,
      bool Optional,
      bool Stalled);

  /// <summary>
  /// Ordered <see cref="CampaignRouteV2.Steps"/> cursor. Advances forward one step at a time whenever
  /// <see cref="AdvanceEngine.IsStepSatisfied"/> is true for the current step, and forward-SNAPS past
  /// any stalled steps on area change (the graceful-degradation path spec §6). Publishes an immutable
  /// <see cref="CampaignStepInstruction"/> snapshot under the same lock/volatile discipline as
  /// <c>RadarApp._campaignGps</c>: writes take <see cref="_lock"/>; readers see a volatile reference
  /// swap so SSE thread never sees a torn record. No memory writes.
  /// </summary>
  public sealed class RouteCursor
  {
      private readonly CampaignRouteV2 _route;
      private readonly object _lock = new();
      private int _ordinal;                                  // guarded by _lock; also read by publisher
      private volatile CampaignStepInstruction[] _published; // 1-slot swap array = torn-read-free reference publication
      private string? _lastArea;                             // world thread only

      public RouteCursor(CampaignRouteV2 route)
      {
          _route = route ?? throw new ArgumentNullException(nameof(route));
          _ordinal = 0;
          _published = _route.Steps.Count > 0
              ? new[] { Snapshot(0, stalled: false) }
              : Array.Empty<CampaignStepInstruction>();
      }

      public int CurrentOrdinal
      {
          get { lock (_lock) return _ordinal; }
      }

      /// <summary>Immutable current-step snapshot for SSE payload. Null when the route is empty or complete.</summary>
      public CampaignStepInstruction? CurrentInstruction
      {
          get { var p = _published; return p.Length == 0 ? null : p[0]; }
      }

      /// <summary>World-thread per-tick advance. Consumes zero heap when nothing changes.</summary>
      public void Tick(IWorldState world)
      {
          lock (_lock)
          {
              if (_ordinal >= _route.Steps.Count) return;
              var advanced = false;
              // Bounded loop: advance while satisfied, stop at first stall — keeps ordinal monotonic within area.
              while (_ordinal < _route.Steps.Count && AdvanceEngine.IsStepSatisfied(_route.Steps[_ordinal], world))
              {
                  _ordinal++;
                  advanced = true;
              }
              if (advanced) Publish(stalled: false);
          }
      }

      /// <summary>Area-change forward-snap (spec §6 graceful degradation). If the current step's
      /// <c>AreaId</c> no longer matches <paramref name="newAreaId"/>, advance to the first step whose
      /// <c>AreaId</c> equals the new area; if none exist ahead, mark stalled and leave ordinal alone.</summary>
      public void OnAreaChange(string newAreaId)
      {
          if (string.IsNullOrEmpty(newAreaId)) return;
          if (string.Equals(newAreaId, _lastArea, StringComparison.OrdinalIgnoreCase)) return;
          _lastArea = newAreaId;
          lock (_lock)
          {
              var snapAt = -1;
              for (var i = _ordinal; i < _route.Steps.Count; i++)
              {
                  if (string.Equals(_route.Steps[i].AreaId, newAreaId, StringComparison.OrdinalIgnoreCase))
                  { snapAt = i; break; }
              }
              if (snapAt >= 0 && snapAt != _ordinal)
              {
                  _ordinal = snapAt;
                  Publish(stalled: false);
              }
              else if (snapAt < 0 && _ordinal < _route.Steps.Count)
              {
                  // No step for this area ahead — surface stalled badge, do not rewind.
                  Publish(stalled: true);
              }
          }
      }

      private void Publish(bool stalled)
      {
          if (_ordinal >= _route.Steps.Count) { _published = Array.Empty<CampaignStepInstruction>(); return; }
          _published = new[] { Snapshot(_ordinal, stalled) };
      }

      private CampaignStepInstruction Snapshot(int ordinal, bool stalled)
      {
          var s = _route.Steps[ordinal];
          return new CampaignStepInstruction(s.Id, s.AreaId, s.Act, ordinal, s.Text, s.Optional, stalled);
      }
  }

  /// <summary>Test-only static hook so unit tests can force <see cref="AdvanceEngine.IsStepSatisfied"/>
  /// deterministic without a full <see cref="IWorldState"/> matrix. Production code never touches this.</summary>
  internal static class AdvanceEngineTestHooks
  {

  [Fact]
  public void ConcurrencyStress_TickAndSnapshotRead_NoTornReadsAndOrdinalMonotonicWithinArea()
  {
      // 200-step single-area route so we can crank Tick as hard as possible without hitting completion too fast.
      var steps = new List<RouteStep>();
      for (var i = 0; i < 200; i++) steps.Add(new($"s{i}", "G1_1", 1, $"step {i}", false, new List<Objective>()));
      var cursor = new RouteCursor(new CampaignRouteV2(2, steps));
      var world  = new FakeWorld { CurrentAreaId = "G1_1" };
      AdvanceEngineTestHooks.SatisfyFor = _ => true;   // every tick advances one

      var stop = false;
      var lastOrdinal = -1;
      var tornRead = false;
      var readerOrdinals = new System.Collections.Concurrent.ConcurrentBag<int>();

      var worldThread = new System.Threading.Thread(() =>
      {
          while (!System.Threading.Volatile.Read(ref stop))
          {
              cursor.Tick(world);
              if (cursor.CurrentOrdinal >= steps.Count) break;
          }
      }) { IsBackground = true };

      var sseThread = new System.Threading.Thread(() =>
      {
          while (!System.Threading.Volatile.Read(ref stop))
          {
              var ins = cursor.CurrentInstruction;
              if (ins is { } snap)
              {
                  // Torn read = Ordinal disagrees with the StepId embedded in the same snapshot.
                  if (snap.StepId != $"s{snap.Ordinal}") tornRead = true;
                  if (snap.Ordinal < lastOrdinal) tornRead = true;   // monotonic-within-area
                  lastOrdinal = snap.Ordinal;
                  readerOrdinals.Add(snap.Ordinal);
              }
          }
      }) { IsBackground = true };

          sseThread.Start();

/* ── director tab: Zone Plan (live ranked queue from /state) + EC2 guide ── */
function renderDirectorQueue(){
  const dq = document.getElementById('dirQueue');
  if (!dq) return;

  const gb = document.getElementById('gpsBanner');
  if (gb) {
    const g = state && state.campaignGps;
    if (g) { gb.hidden = false; gb.textContent = '🧭 ' + g; }
    else { gb.hidden = true; gb.textContent = ''; }
  }

  // EC2 CampaignGuide (v0.21+, additive; may be null on v0.20 backends or when off)
  const guide = state && state.campaignGuide;
  const stepEl = document.getElementById('guideStep');
  const badgeEl = document.getElementById('guideDegradeBadge');
  if (stepEl && badgeEl) {
    if (guide && guide.available && guide.stepText) {
      stepEl.hidden = false;
      stepEl.textContent = '▶ ' + guide.stepText;
      badgeEl.hidden = !guide.degraded;
    } else {
      stepEl.hidden = true;
      stepEl.textContent = '';
      badgeEl.hidden = true;
    }
  }

  const dir = (state && state.director) || [];
  if (dir.length === 0){
    dq.innerHTML = '<div style="opacity:.5;padding:4px 0">No active objectives in this zone</div>';
    return;
  }
  dq.innerHTML = dir.map((o, i) =>
    `<div class="navrow" style="font-weight:${i===0?'600':'400'};opacity:${i===0?'1':'.75'}">` +
    `${i===0 ? '&#9654; ' : ''}` +
    `<span class="navname">${esc(o.label)}</span>` +
    `<span class="navtag">${esc(o.tier||o.category)}</span>` +
    `<span class="navdist">P${o.priority}</span>` +
    `</div>`
  ).join('');
}

  <#
  .SYNOPSIS
    Attribution sentinel gate for the EC2 (ExileCampaigns2) port.
    Mode Draft   → this task. Enforces DRAFT-phase invariants:
      1. Every "sentinel-required" file contains BOTH `TODO(syrairc-license)`
         and `TODO(syrairc-hash)` at least once.
      2. No touched public surface contains a bare `<license>` or `<hash>`
         angle-bracket token (case-insensitive).
      3. No touched public surface contains any `superpowers/`, `.superpowers/`,
         or `docs/superpowers/` path substring — internal-tooling never leaks.
    Mode Formalize → flipped by EC2-ATTR-FORMALIZE. Rejects any surviving
      `TODO(syrairc-license)` or `TODO(syrairc-hash)` sentinel.

    Exit 0 = pass. Exit 1 = fail (with per-hit listing).
  #>
  [CmdletBinding()]
  param(
    [string]$Root,
    [ValidateSet('Draft','Formalize')][string]$Mode = 'Draft'
  )
  $ErrorActionPreference = 'Stop'
  $scriptDir = if ($PSScriptRoot) { $PSScriptRoot } else { Split-Path -Parent $MyInvocation.MyCommand.Path }
  if (-not $Root) { $Root = Split-Path -Parent $scriptDir }
  $rootFull = (Resolve-Path -LiteralPath $Root).Path

  # Files that MUST carry both sentinels in DRAFT mode.
  $sentinelRequired = @(
    'README.md',
    'CHANGELOG.md',
    'src/POE2Radar.Core/Campaign/Guide/Data/poe2/HEADER.md',
    'src/POE2Radar.Core/Campaign/Guide/RouteModel.cs',
    'src/POE2Radar.Core/Campaign/Guide/AdvanceEngine.cs',
    'src/POE2Radar.Core/Campaign/Guide/StepMeta.cs',
    'src/POE2Radar.Core/Campaign/Guide/PatternMatcher.cs',
    'src/POE2Radar.Overlay/Web/DashboardHtml.cs',
    'src/POE2Radar.Overlay/Web/ApiServer.cs',
    'scratchpad/discord-v0.21.md'
  )
  # Files scanned for bare tokens AND superpowers leakage (public-surface set).
  $publicSurfaces = @(
    'README.md',
    'CHANGELOG.md',
    'CONTRIBUTING.md',
    'docs/community-pipeline.md',
    'src/POE2Radar.Core/Campaign/Guide/Data/poe2/HEADER.md',
    'scratchpad/discord-v0.21.md'
  )
  # Plus every .github/ISSUE_TEMPLATE/*.yml and every DashboardHtml/ApiServer .cs
  $globs = @(
    '.github/ISSUE_TEMPLATE/*.yml',
    'src/POE2Radar.Overlay/Web/DashboardHtml.cs',
    'src/POE2Radar.Overlay/Web/ApiServer.cs',
    'cloudflare-worker/**/*.html',
    'cloudflare-worker/**/*.js'
  )

  $fails = New-Object System.Collections.ArrayList
  function Fail([string]$msg) { [void]$fails.Add($msg) }

  # 1. sentinel-required files
  foreach ($rel in $sentinelRequired) {
    $p = Join-Path $rootFull $rel
    if (-not (Test-Path -LiteralPath $p)) {
      if ($Mode -eq 'Draft') { Fail "missing file (sentinel-required): $rel" }
      continue
    }
    $text = Get-Content -LiteralPath $p -Raw
    if ($Mode -eq 'Draft') {
      if ($text -notmatch [regex]::Escape('TODO(syrairc-license)')) { Fail "missing sentinel TODO(syrairc-license) in $rel" }
      if ($text -notmatch [regex]::Escape('TODO(syrairc-hash)'))    { Fail "missing sentinel TODO(syrairc-hash) in $rel" }
    } else {
      if ($text -match [regex]::Escape('TODO(syrairc-license)')) { Fail "surviving sentinel TODO(syrairc-license) in $rel (Formalize)" }
      if ($text -match [regex]::Escape('TODO(syrairc-hash)'))    { Fail "surviving sentinel TODO(syrairc-hash) in $rel (Formalize)" }
    }
  }

  # 2 + 3. bare-token + superpowers scan across the public-surface set
  $scanFiles = @()
  foreach ($rel in $publicSurfaces) {
    $p = Join-Path $rootFull $rel
    if (Test-Path -LiteralPath $p) { $scanFiles += (Get-Item -LiteralPath $p) }
  }
  foreach ($g in $globs) {
    $matches = Get-ChildItem -Path (Join-Path $rootFull $g) -File -ErrorAction SilentlyContinue
    if ($matches) { $scanFiles += $matches }
  }

  $bareLicense  = '(?i)(?<![A-Za-z_])<license>(?![A-Za-z_])'
  $bareHash     = '(?i)(?<![A-Za-z_])<hash>(?![A-Za-z_])'
  $superpowers  = '(?i)(?:^|[\s"''`(/])(?:\.?superpowers/|docs/superpowers/)'

  foreach ($f in $scanFiles) {
    $rel = ($f.FullName.Substring($rootFull.Length).TrimStart('\','/')) -replace '\\','/'
    $n = 0
    foreach ($line in [System.IO.File]::ReadLines($f.FullName)) {
      $n++
      if ($line -match $bareLicense)  { Fail "bare token <license> in $rel`:$n  $($line.Trim())" }
      if ($line -match $bareHash)     { Fail "bare token <hash> in $rel`:$n  $($line.Trim())" }
      if ($line -match $superpowers)  { Fail "superpowers path in public surface $rel`:$n  $($line.Trim())" }
    }
  }

  if ($fails.Count -gt 0) {
    Write-Host "ATTRIBUTION-SENTINEL-GATE ($Mode): FAIL" -ForegroundColor Red
    foreach ($m in $fails) { Write-Host "  $m" }
    exit 1
  }
  Write-Host "ATTRIBUTION-SENTINEL-GATE ($Mode): PASS" -ForegroundColor Green
  exit 0
  ```

- [ ] **Step 6: Create the reusable pre-check script `scripts/tag-precheck.ps1`**

  ```powershell
  # scripts/tag-precheck.ps1
  <#
  .SYNOPSIS
    Rejects any surviving TODO(syrairc-*) sentinel in shipped surfaces.
    Callable from CI (Mode=CI) and from LO's release routine before `git tag` (Mode=Tag).
  #>
  [CmdletBinding()]
  param(
    [string]$Root,
    [ValidateSet('CI','Tag','SelfTest')] [string]$Mode = 'Tag'
  )
  $ErrorActionPreference = 'Stop'
  $scriptDir = if ($PSScriptRoot) { $PSScriptRoot } else { Split-Path -Parent $MyInvocation.MyCommand.Path }
  if (-not $Root) { $Root = Split-Path -Parent $scriptDir }
  $rootFull = (Resolve-Path -LiteralPath $Root).Path

  # Allowlist: spec/history docs may legitimately mention the sentinel literal.
  $allowRegex = '[\\/](docs[\\/]superpowers|scratchpad[\\/]discord-v0\.(19|20)\.)'

  if ($Mode -eq 'SelfTest') {
    $tmp = Join-Path ([IO.Path]::GetTempPath()) ('tagcheck-' + [guid]::NewGuid())
    New-Item -ItemType Directory -Path $tmp | Out-Null
    Set-Content (Join-Path $tmp 'bad.md')  'header: TODO(syrairc-license) footer'
    Set-Content (Join-Path $tmp 'good.md') 'header: MIT footer'
    $bad = Select-String -Path (Join-Path $tmp '*.md') -Pattern 'TODO\(syrairc-' |
           Where-Object { $_.Path -notmatch $allowRegex }
    Remove-Item $tmp -Recurse -Force
    if ($bad.Count -ne 1) { Write-Error "self-test FAILED: expected 1 hit, got $($bad.Count)"; exit 2 }
    Write-Host 'tag-precheck self-test PASSED.' -ForegroundColor Green
    exit 0
  }

  $exts = @('*.md','*.cs','*.ps1','*.yml','*.yaml','*.json','*.html','*.js','*.py','*.txt')
  $files = foreach ($e in $exts) {
    Get-ChildItem -Path $rootFull -Recurse -File -Filter $e -ErrorAction SilentlyContinue |
      Where-Object { $_.FullName -notmatch '[\\/](bin|obj|node_modules|\.git)[\\/]' -and $_.FullName -notmatch $allowRegex }
  }
  if (-not $files) { Write-Host 'tag-precheck: no candidate files (unexpected).'; exit 0 }

  $hits = @(Select-String -Path $files.FullName -Pattern 'TODO\(syrairc-' -SimpleMatch:$false)
  if ($hits.Count -gt 0) {
    Write-Host "TAG PRE-CHECK: FAIL - $($hits.Count) surviving TODO(syrairc-*) sentinel(s):" -ForegroundColor Red
    foreach ($h in $hits) {
      $rel = $h.Path.Substring($rootFull.Length).TrimStart('\','/').Replace('\','/')
      Write-Host ("  {0}:{1}  {2}" -f $rel, $h.LineNumber, $h.Line.Trim())
    }
    if ($Mode -eq 'Tag') { Write-Host 'BLOCKING tag. Resolve syrairc DM (PMS-12) and re-run EC2-ATTR-FORMALIZE.' -ForegroundColor Red }
    exit 1
  }
  Write-Host "TAG PRE-CHECK: PASS - no TODO(syrairc-*) sentinels in shipped surfaces (Mode=$Mode)." -ForegroundColor Green
  exit 0
  ```

  ```javascript
  // POE2GPS community-pack collector (v3). Splits the v2 single-route into three sibling routes:
  //   POST /submit-atlas   — {names, objectives}   (v0.20 backward-compat shape)
  //   POST /submit-buffs   — {buffs}               (buff metadata paths + tier)
  //   POST /submit-preload — {preloads}            (metadata paths; bare .dds/.ao rejected)
  // Shared middleware: NFKD+leet profanity fold, KV rate limit 5/60s per CF-Connecting-IP, GH Issue dispatch.
  // GITHUB_TOKEN is a Worker secret; never in the client. RATE_KV is a KV namespace bound in wrangler.toml.

  const REPO = 'luther-rotmg/POE2GPS';
  const MAX_BYTES = 262144; // 256 KB
  const RATE_LIMIT = 5;
  const RATE_WINDOW_S = 60;
  const CORS = {
    'Access-Control-Allow-Origin': '*',
    'Access-Control-Allow-Methods': 'POST, OPTIONS',
    'Access-Control-Allow-Headers': 'Content-Type',
  };
  const SLURS = ['nigger', 'faggot', 'retard', 'kike', 'spic', 'chink', 'cunt'];

  // ─── shared middleware: profanity (NFKD + strip combining marks + leet-fold + collapse) ───
  const LEET = { '4': 'a', '@': 'a', '0': 'o', '1': 'i', '3': 'e', '5': 's', '$': 's', '7': 't', '8': 'b' };
  export function isProfane(s) {
    if (typeof s !== 'string') return false;
    // NFKD normalize, strip combining marks (U+0300-U+036F), leet-fold, drop non-letters, lowercase
    const folded = s
      .normalize('NFKD')
      .replace(/[̀-ͯ]/g, '')
      .toLowerCase()
      .split('')
      .map(ch => LEET[ch] ?? ch)
      .join('')
      .replace(/[^a-z]+/g, '');
    // Also fold 1 -> l (some slurs use l), pass 2 over the collapsed string
    const alt = folded.replace(/i/g, 'l');
    return SLURS.some(w => folded.includes(w) || alt.includes(w));
  }

  // ─── shared middleware: KV rate limit (5 per 60s per IP) ───
  export async function rateLimit(env, ip) {
    if (!env || !env.RATE_KV || !ip) return true; // fail-open on missing binding (tests without KV, dev)
    const key = `rl:${ip}`;
    const raw = await env.RATE_KV.get(key);
    const n = raw ? parseInt(raw, 10) : 0;
    if (n >= RATE_LIMIT) return false;
    await env.RATE_KV.put(key, String(n + 1), { expirationTtl: RATE_WINDOW_S });
    return true;
  }

  // ─── routing ───
  export function routeFor(url) {
    switch (url.pathname) {
      case '/submit-atlas':   return { kind: 'atlas' };
      case '/submit-buffs':   return { kind: 'buffs' };
      case '/submit-preload': return { kind: 'preload' };
      default:                return null;
    }
  }

  // ─── shared identifying-field reject ───
  const FORBIDDEN = ['charname', 'character', 'account', 'accountname', 'address', 'ip'];
  function hasIdentifying(pack) {
    const s = JSON.stringify(pack).toLowerCase();
    return FORBIDDEN.some(k => s.includes('"' + k + '"'));
  }

  function cleanLabel(raw, max) {
    if (typeof raw !== 'string') return null;
    const s = raw.trim();
    if (s.length < 2 || s.length > max) return null;
    if (!/[a-zA-Z]/.test(s)) return null;
    return s;
  }

  // ─── per-route payload filter (also exported so tests can hit filterPayloadCommon('preload', ...)) ───
  export function filterPayloadCommon(kind, pack) {
    if (kind === 'atlas') {
      if (!pack || typeof pack.names !== 'object' || Array.isArray(pack.names) || !Array.isArray(pack.objectives))
        return { error: 'expected {names:object, objectives:array}' };
      const seen = new Set();
      const names = [];
      for (const [meta, raw] of Object.entries(pack.names)) {
        const nm = cleanLabel(raw, 60);
        if (!nm || isProfane(nm)) continue;
        const key = String(meta).slice(0, 200);
        if (seen.has(key)) continue;
        seen.add(key);
        names.push([key, nm]);
      }
      const objectives = [];
      for (const o of pack.objectives) {
        if (!o || typeof o !== 'object') continue;
        const label = cleanLabel(o.label, 60);
        const category = cleanLabel(o.category, 40);
        if (!label && !category) continue;
        if ((label && isProfane(label)) || (category && isProfane(category))) continue;
        objectives.push({
          label, category,
          priority: o.priority, enabled: o.enabled,
          metadata: o.metadata, categories: o.categories,
          poi: o.poi, rarity: o.rarity, landmarkPath: o.landmarkPath,
        });
      }
      return { names, objectives };
    }
    if (kind === 'buffs') {
      if (!pack || !Array.isArray(pack.buffs)) return { error: 'expected {buffs:array}' };
      const buffs = [];
      for (const b of pack.buffs) {
        if (!b || typeof b !== 'object') continue;
        const path = cleanLabel(b.path, 200);
        if (!path || isProfane(path)) continue;
        const tier = (typeof b.tier === 'string' || typeof b.tier === 'number') ? b.tier : null;
        buffs.push({ path, tier, metadata: b.metadata });
      }
      return { buffs };
    }
    if (kind === 'preload') {
      if (!pack || !Array.isArray(pack.preloads)) return { error: 'expected {preloads:array}' };
      const preloads = [];
      for (const p of pack.preloads) {
        if (!p || typeof p !== 'object') continue;
        const path = cleanLabel(p.path, 200);
        if (!path || isProfane(path)) continue;
        // Reject bare .dds / .ao (raw asset filenames with no metadata prefix)
        if (/^[^\/]+\.(dds|ao)$/i.test(path)) continue;
        const freq = (typeof p.freq === 'number' && Number.isFinite(p.freq)) ? p.freq : null;
        preloads.push({ path, freq });
      }
      return { preloads };
    }
    return { error: 'unknown route' };
  }

  // ─── issue builder — sub-label per route ───
  function buildIssue(kind, f) {
    const baseLabels = ['community-pack', 'needs-review'];
    if (kind === 'atlas') {
      const cats = [...new Set(f.objectives.map(o => o.category).filter(Boolean))].sort();
      const sample = f.names.slice(0, 10).map(([, v]) => '- `' + v + '`').join('\n');
      const full = JSON.stringify({ names: Object.fromEntries(f.names), objectives: f.objectives }, null, 2).slice(0, MAX_BYTES);
      const body = [
        `**${f.names.length} names, ${f.objectives.length} objectives** (auto-filtered)`,
        cats.length ? 'Categories used: ' + cats.map(c => '`' + c + '`').join(', ') : '',
        f.names.length ? '\nSample names:\n' + sample : '',
        '\n<details><summary>Full pack JSON</summary>\n\n```json\n' + full + '\n```\n</details>',
        '\n*Review, then label `approved` to fold into the next release (or close to reject).*',
      ].filter(Boolean).join('\n');
      return { title: `Community pack: ${f.names.length} names, ${f.objectives.length} objectives`, body, labels: baseLabels };
    }
    if (kind === 'buffs') {
      const full = JSON.stringify(f, null, 2).slice(0, MAX_BYTES);
      const body = `**${f.buffs.length} buff paths** (auto-filtered)\n\n<details><summary>Full pack JSON</summary>\n\n\`\`\`json\n${full}\n\`\`\`\n</details>\n\n*Review, then label \`approved\` to fold into BuffCatalog.*`;
      return { title: `Community pack (buffs): ${f.buffs.length} paths`, body, labels: [...baseLabels, 'buffs'] };
    }
    if (kind === 'preload') {
      const full = JSON.stringify(f, null, 2).slice(0, MAX_BYTES);
      const body = `**${f.preloads.length} preload paths** (auto-filtered; bare .dds/.ao rejected)\n\n<details><summary>Full pack JSON</summary>\n\n\`\`\`json\n${full}\n\`\`\`\n</details>\n\n*Review, then label \`approved\` to fold into PreloadCatalog.*`;
      return { title: `Community pack (preload): ${f.preloads.length} paths`, body, labels: [...baseLabels, 'preload'] };
    }
  }

  function json(status, obj) {
    return new Response(JSON.stringify(obj), { status, headers: { 'Content-Type': 'application/json', ...CORS } });
  }

  export default {
    async fetch(request, env) {
      if (request.method === 'OPTIONS') return new Response(null, { headers: CORS });
      if (request.method !== 'POST')    return json(405, { error: 'method not allowed' });

      const url = new URL(request.url);
      const route = routeFor(url);
      if (!route) return json(404, { error: 'unknown route' });

      const ip = request.headers.get('CF-Connecting-IP') || 'unknown';
      if (!(await rateLimit(env, ip))) return json(429, { error: 'rate limit exceeded' });

      const body = await request.text();
      if (new TextEncoder().encode(body).length > MAX_BYTES) return json(413, { error: 'payload too large' });

      let pack;
      try { pack = JSON.parse(body); } catch { return json(400, { error: 'invalid json' }); }
      if (!pack || typeof pack !== 'object') return json(400, { error: 'invalid pack' });
      if (hasIdentifying(pack)) return json(400, { error: 'identifying field present' });

      const f = filterPayloadCommon(route.kind, pack);
      if (f.error) return json(400, { error: f.error });
      const nonEmpty = (route.kind === 'atlas'   && (f.names.length || f.objectives.length))
                    || (route.kind === 'buffs'   && f.buffs.length)
                    || (route.kind === 'preload' && f.preloads.length);
      if (!nonEmpty) return json(400, { error: 'nothing valid after filtering' });

      const gh = await fetch(`https://api.github.com/repos/${REPO}/issues`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${env.GITHUB_TOKEN}`,
          'Accept': 'application/vnd.github+json',
          'User-Agent': 'poe2gps-contribute-worker',
          'Content-Type': 'application/json',
        },
        body: JSON.stringify(buildIssue(route.kind, f)),
      });
      if (!gh.ok) return json(502, { error: 'github rejected', status: gh.status });

      const accepted = route.kind === 'atlas'
        ? { names: f.names.length, objectives: f.objectives.length }
        : route.kind === 'buffs' ? { buffs: f.buffs.length } : { preloads: f.preloads.length };
      return json(200, { ok: true, accepted });
    },
  };
  ```

/// <summary>Rewrites a Contribute worker URL onto a sibling route. Accepts base URL or any existing
/// /submit-* suffix; strips a trailing slash; either replaces the terminal /submit-* segment or
/// appends /submit-&lt;sibling&gt; when none is present. Internal for testability.</summary>
internal static string SiblingContributeUrl(string url, string sibling)
{
    var trimmed = (url ?? "").TrimEnd('/');
    var re = new System.Text.RegularExpressions.Regex(@"/submit-[a-z]+$", System.Text.RegularExpressions.RegexOptions.CultureInvariant);
    return re.IsMatch(trimmed) ? re.Replace(trimmed, "/submit-" + sibling) : trimmed + "/submit-" + sibling;
}

#!/usr/bin/env python3
"""Fold APPROVED community-pack issues into the embedded tables + sidecars.

Maintainer ritual (per release): review the `community-pack` + `needs-review` issues, label the good ones
`approved` (or close to reject), then run:

    python resources/poe2-data/merge_community.py [--dry-run] [--preload]

Pulls APPROVED issues via `gh` (body + author + number + url so we can emit a credit block),
parses the pack JSON from each issue body, and folds:
  - kind absent OR "atlas"  -> names -> entity_names.json (curated wins); novel categories -> labels.json.
  - kind "preload"          -> paths + hit counts -> preload_catalog_community.json sidecar
                               (consumed by v0.22 PreloadCatalog seed loader; ships now to unblock CF-DASH-BUTTONS).
  - kind "buffs"            -> DEFERRED to v0.22 (SL #9 TieredCatalog<T> shipping-gate not yet green).

Read-only / data-only.
"""
from __future__ import annotations

import json
import re
import subprocess
import sys
from pathlib import Path

REPO = "luther-rotmg/POE2GPS"
ENTITY_NAMES = Path("src/POE2Radar.Core/Game/entity_names.json")
LABELS = Path("src/POE2Radar.Overlay/Web/labels.json")
PRELOAD_SIDECAR = Path("src/POE2Radar.Core/Game/preload_catalog_community.json")


def normalize_key(raw: str) -> str:
    k = raw.strip()
    at = k.find("@")
    if at >= 0:
        k = k[:at]
    return k.lower()


def fetch_approved_issues() -> list[dict]:
    out = subprocess.run(
        ["gh", "issue", "list", "-R", REPO, "--label", "community-pack", "--label", "approved",
         "--state", "all", "--limit", "1000", "--json", "body,author,number,url"],
        capture_output=True, text=True, check=True).stdout
    return json.loads(out)


def extract_pack(body: str) -> dict | None:
    m = re.search(r"```json\s*(\{.*?\})\s*```", body, re.S)
    if not m:
        return None
    try:
        return json.loads(m.group(1))
    except json.JSONDecodeError:
        return None


def fold_atlas(pack: dict, names: dict[str, str], cats: set[str]) -> None:
    for k, v in (pack.get("names") or {}).items():
        if isinstance(v, str) and v.strip():
            names[normalize_key(k)] = v.strip()
    for o in (pack.get("objectives") or []):
        c = (o.get("category") or "").strip() if isinstance(o, dict) else ""
        if c:
            cats.add(c)


def fold_preload(pack: dict, hits: dict[str, int]) -> None:
    p = pack.get("preload") or {}
    src = p.get("freq") if isinstance(p, dict) and isinstance(p.get("freq"), dict) else p
    if not isinstance(src, dict):
        return
    for path, count in src.items():
        if not isinstance(path, str) or not path.strip():
            continue
        try:
            n = int(count)
        except (TypeError, ValueError):
            continue
        if n <= 0:
            continue
        key = path.strip().lower()
        hits[key] = hits.get(key, 0) + n


def emit_credit_block(contributors: list[tuple[str, int, str]]) -> str:
    lines = ["### Special thanks — community-pack contributors", ""]
    for handle, num, url in sorted(set(contributors)):
        lines.append(f"- @{handle} — #{num} ({url})")
    return "\n".join(lines) + "\n"


def main(argv: list[str]) -> int:
    dry = "--dry-run" in argv
    do_preload = "--preload" in argv
    issues = fetch_approved_issues()
    names: dict[str, str] = {}
    cats: set[str] = set()
    hits: dict[str, int] = {}
    contributors: list[tuple[str, int, str]] = []
    packs = 0
    for issue in issues:
        body = issue.get("body") or ""
        pack = extract_pack(body)
        if not pack:
            continue
        packs += 1
        author = ((issue.get("author") or {}).get("login") or "").strip()
        number = int(issue.get("number") or 0)
        url = (issue.get("url") or "").strip()
        if author and number and url:
            contributors.append((author, number, url))

        kind = (pack.get("kind") or "atlas").strip().lower()
        if kind in ("", "atlas"):
            fold_atlas(pack, names, cats)
        elif kind == "preload" and do_preload:
            fold_preload(pack, hits)
        elif kind == "buffs":
            # SL #10 buff-fold DEFERRED to v0.22 — TieredCatalog<T> shipping gate (SL #9) still red.
            print(f"skip: buffs-kind pack in issue #{number} — buff-fold deferred to v0.22 (SL #9 gate red)")

    # Atlas fold (curated wins).
    table = json.loads(ENTITY_NAMES.read_text(encoding="utf-8-sig"))
    added = 0
    for k, v in names.items():
        if k not in table:
            table[k] = v
            added += 1
    labels = json.loads(LABELS.read_text(encoding="utf-8-sig"))
    existing = {l for arr in labels.values() for l in arr}
    novel = sorted(c for c in cats if c not in existing)
    if novel:
        labels.setdefault("Community", [])
        for c in novel:
            if c not in labels["Community"]:
                labels["Community"].append(c)
    print(f"{packs} approved pack(s): +{added} names, novel labels: {novel or 'none'}")

    # Preload sidecar (SL #11 ships now).
    if do_preload and hits:
        sidecar = {"paths": [{"path": p, "count": n} for p, n in sorted(hits.items())]}
        print(f"preload fold: {len(hits)} unique paths (total hits {sum(hits.values())})")
        if not dry:
            PRELOAD_SIDECAR.parent.mkdir(parents=True, exist_ok=True)
            PRELOAD_SIDECAR.write_text(
                json.dumps(sidecar, ensure_ascii=False, indent=2),
                encoding="utf-8")
            print(f"wrote {PRELOAD_SIDECAR}")

  # test-relabel-atlas-issues.sh — harness for relabel-atlas-issues.sh
  # Shims `gh` via a bash function so we can drive the script offline.
  set -euo pipefail

  HERE="$(cd "$(dirname "$0")" && pwd)"
  SCRIPT="$HERE/relabel-atlas-issues.sh"
  WORK="$(mktemp -d)"
  trap 'rm -rf "$WORK"' EXIT

  # State file: JSON array of {number, labels}
  STATE="$WORK/state.json"
  cat > "$STATE" <<'JSON'
  [
    {"number": 101, "labels": ["atlas-submission"]},
    {"number": 102, "labels": ["atlas-submission", "community-pack"]},
    {"number": 103, "labels": ["atlas-submission", "needs-triage"]}
  ]
  JSON

  # gh shim: emulates the two subcommands the script uses.
  gh() {
    case "$1 $2" in
      "issue list")
        # Emit JSON matching --label atlas-submission --state open --json number
        jq '[.[] | select(.labels | index("atlas-submission")) | {number}]' "$STATE"
        ;;
      "issue edit")
        local num="$3"
        # Parse --add-label VALUE
        shift 3
        local label=""
        while [ $# -gt 0 ]; do
          if [ "$1" = "--add-label" ]; then label="$2"; shift 2; else shift; fi
        done
        # Mutate state: append label if not already present.
        jq --argjson n "$num" --arg L "$label" \
          '(.[] | select(.number == $n) | .labels) |= (. + [$L] | unique)' \
          "$STATE" > "$STATE.tmp" && mv "$STATE.tmp" "$STATE"
        ;;
      *) echo "gh shim: unhandled: $*" >&2; return 2 ;;
    esac
  }
  export -f gh

  # First run
  bash "$SCRIPT"
  first="$(cat "$STATE")"

  # Every atlas-submission issue must now carry community-pack
  missing="$(jq '[.[] | select(.labels | index("atlas-submission")) | select(.labels | index("community-pack") | not)] | length' <<<"$first")"
  if [ "$missing" != "0" ]; then
    echo "FAIL: $missing atlas-submission issues missing community-pack" >&2
    exit 1
  fi

  # Second run — idempotent
  bash "$SCRIPT"
  second="$(cat "$STATE")"
  if [ "$first" != "$second" ]; then
    echo "FAIL: second run mutated state (non-idempotent)" >&2
    diff <(echo "$first") <(echo "$second") >&2
    exit 1
  fi

  dupes="$(jq '[.[] | .labels | (length - (unique | length))] | add' <<<"$second")"
  if [ "$dupes" != "0" ]; then

def fetch_approved_issues() -> list[dict]:
    """Fetch APPROVED community-pack issues with body + author + number + url.

    Returns raw gh dicts. `author` may be None (deleted account / bot).
    """
    out = subprocess.run(
        ["gh", "issue", "list", "-R", REPO, "--label", "community-pack", "--label", "approved",
         "--state", "all", "--limit", "1000", "--json", "body,author,number,url"],
        capture_output=True, text=True, check=True).stdout
    return json.loads(out)


def build_credit_block(contributors: list[tuple[str, list[tuple[int, str]]]]) -> str:
    """Deterministic markdown credit block.

    `contributors` is a list of (handle, [(issue_number, issue_url), ...]) tuples.
    Handles are sorted case-insensitively; issues per handle are sorted ascending
    by number. Rendered as a `### Community contributors` H3 block plus one bullet
    per contributor, ready to paste into CHANGELOG.md above the themed body.
    """
    lines = ["### Community contributors", ""]
    for handle, issues in contributors:
        refs = ", ".join(f"[#{n}]({u})" for n, u in issues)
        lines.append(f"- @{handle} — {refs}")
    lines.append("")
    return "\n".join(lines)
```

def main(argv: list[str]) -> int:
    dry = "--dry-run" in argv
    issues = fetch_approved_issues()

    names: dict[str, str] = {}
    cats: set[str] = set()
    # handle -> dict[int, str]  (number -> url), dedup'd per-issue
    credit: dict[str, dict[int, str]] = {}
    packs = 0

    for issue in issues:
        body = issue.get("body") or ""
        pack = extract_pack(body)
        if not pack:
            continue
        packs += 1
        for k, v in (pack.get("names") or {}).items():
            if isinstance(v, str) and v.strip():
                names[normalize_key(k)] = v.strip()
        for o in (pack.get("objectives") or []):
            c = (o.get("category") or "").strip() if isinstance(o, dict) else ""
            if c:
                cats.add(c)

        author = issue.get("author") or {}
        handle = (author.get("login") if isinstance(author, dict) else None) or "unknown"
        number = int(issue.get("number") or 0)
        url = issue.get("url") or ""
        if number:
            credit.setdefault(handle, {})[number] = url

    # Fold names (curated wins).
    table = json.loads(ENTITY_NAMES.read_text(encoding="utf-8-sig"))
    added = 0
    for k, v in names.items():
        if k not in table:
            table[k] = v
            added += 1

    # Fold novel categories into labels.json under a "Community" group.
    labels = json.loads(LABELS.read_text(encoding="utf-8-sig"))
    existing = {l for arr in labels.values() for l in arr}
    novel = sorted(c for c in cats if c not in existing)
    if novel:
        labels.setdefault("Community", [])
        for c in novel:
            if c not in labels["Community"]:
                labels["Community"].append(c)

    print(f"{packs} approved pack(s): +{added} names, novel labels: {novel or 'none'}")

    # Deterministic credit block: handles sorted case-insensitively, issues per
    # handle sorted ascending by number. Rerunning against the same issue set
    # produces byte-identical output.
    contributors = [
        (h, sorted(nu.items()))
        for h, nu in sorted(credit.items(), key=lambda kv: kv[0].lower())
    ]
    if contributors:
        print()
        print("--- credit block (paste above the themed body in CHANGELOG.md) ---")
        print(build_credit_block(contributors))
        print("--- end credit block ---")

    if dry:
        print("(dry-run — no files written)")
        return 0
    ENTITY_NAMES.write_text(
        json.dumps(table, ensure_ascii=False, sort_keys=True, separators=(",", ":")),
        encoding="utf-8")
    LABELS.write_text(json.dumps(labels, ensure_ascii=False, indent=2), encoding="utf-8")
    print(f"wrote {ENTITY_NAMES} and {LABELS}")
    return 0
